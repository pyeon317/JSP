-- TABLE
CREATE TABLE USERS (
    ID VARCHAR2(30) ,
    PW VARCHAR2(30) NOT NULL,
    NAME VARCHAR2(30) NOT NULL,
    EMAIL VARCHAR2(30),
    GENDER CHAR(1),
    REGDATE DATE DEFAULT SYSDATE
);
ALTER TABLE USERS ADD CONSTRAINT USERS_PK PRIMARY KEY (ID);
ALTER TABLE USERS ADD CONSTRAINT USERS_CHECK CHECK( GENDER IN ('M', 'F') );
